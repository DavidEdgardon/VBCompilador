Public module modulo1
    sub main()
	Dim x as integer
        Dim y as integer
	dim n as Integer 
	dim i As Integer 
	dim sum As Integer
	dim val As Integer
	
	console.writeline("Ingrese numero para calcular Fibonacci")
	val = console.readline()
	n = Val
	x = 0
	y = 1

	For i = 3 To n step 1
	sum = x + y
	console.writeline(sum)
	x = y
	y = sum
	y = sum
	Next i

    End Sub
end module
